def climbingLeaderboard(ranked, player)
    player_ranks = []
    i = 0
    while i  len(player)
        p = player[i]
        rank = 1
        prev = None
        j = 0
        while j  len(ranked)
            r = ranked[j]
            if r != prev
                if p  r
                    rank = rank+1
                else
                    break
                prev = r
            j = j+1
        player_ranks.append(rank)
        i = i+1
    return player_ranks
