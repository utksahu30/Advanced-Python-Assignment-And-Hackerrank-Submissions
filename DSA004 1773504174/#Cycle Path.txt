# Complete the has_cycle function below.

#
# For your reference:
#
# SinglyLinkedListNode:
#     int data
#     SinglyLinkedListNode next
#
#

def has_cycle(head):
    visited=[]
    current= head
    while current is not None:
        if current in visited:
            return 1
        visited.append(current)
        
        current= current.next
    return 0
